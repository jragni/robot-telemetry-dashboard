**Bottom Tab Bar tests:**

```typescript
test.describe('Bottom Tab Bar', () => {
  test('shows bottom tab bar on mobile viewport', async ({ page }) => {
    await page.setViewportSize({ width: 375, height: 812 });
    await page.goto('/robot-telemetry-dashboard/dashboard');
    // Expect bottom nav to exist with 4 tabs
    await expect(
      page.getByRole('navigation', { name: /bottom/i })
    ).toBeVisible();
    await expect(page.getByRole('link', { name: /dashboard/i })).toBeVisible();
    await expect(page.getByRole('link', { name: /fleet/i })).toBeVisible();
    await expect(page.getByRole('link', { name: /map/i })).toBeVisible();
    await expect(page.getByRole('link', { name: /pilot/i })).toBeVisible();
  });

  test('hides bottom tab bar on desktop viewport', async ({ page }) => {
    await page.setViewportSize({ width: 1440, height: 900 });
    await page.goto('/robot-telemetry-dashboard/dashboard');
    await expect(
      page.getByRole('navigation', { name: /bottom/i })
    ).not.toBeVisible();
  });

  test('highlights active tab', async ({ page }) => {
    await page.setViewportSize({ width: 375, height: 812 });
    await page.goto('/robot-telemetry-dashboard/dashboard');
    const dashTab = page.getByRole('link', { name: /dashboard/i });
    await expect(dashTab).toHaveAttribute('aria-current', 'page');
  });

  test('navigates between tabs', async ({ page }) => {
    await page.setViewportSize({ width: 375, height: 812 });
    await page.goto('/robot-telemetry-dashboard/dashboard');
    await page.getByRole('link', { name: /fleet/i }).click();
    await expect(page).toHaveURL(/\/fleet/);
    await page.getByRole('link', { name: /map/i }).click();
    await expect(page).toHaveURL(/\/map/);
  });
});
```

**Responsive Transition tests:**

```typescript
test.describe('Responsive Transitions', () => {
  test('desktop to mobile: bottom bar appears, grid switches to carousel', async ({
    page,
  }) => {
    await page.setViewportSize({ width: 1440, height: 900 });
    await page.goto('/robot-telemetry-dashboard/dashboard');
    await expect(
      page.getByRole('navigation', { name: /bottom/i })
    ).not.toBeVisible();

    await page.setViewportSize({ width: 375, height: 812 });
    await page.waitForTimeout(500);
    await expect(
      page.getByRole('navigation', { name: /bottom/i })
    ).toBeVisible();
  });

  test('mobile to desktop: bottom bar disappears, carousel switches to grid', async ({
    page,
  }) => {
    await page.setViewportSize({ width: 375, height: 812 });
    await page.goto('/robot-telemetry-dashboard/dashboard');
    await expect(
      page.getByRole('navigation', { name: /bottom/i })
    ).toBeVisible();

    await page.setViewportSize({ width: 1440, height: 900 });
    await page.waitForTimeout(500);
    await expect(
      page.getByRole('navigation', { name: /bottom/i })
    ).not.toBeVisible();
  });
});
```
