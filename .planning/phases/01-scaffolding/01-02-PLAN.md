`src/__tests__/roslib-smoke.test.ts`:
```typescript
import { describe, it, expect } from 'vitest';

describe('roslib CJS/ESM interop', () => {
  it('imports roslib without errors and exposes Ros constructor', async () => {
    const ROSLIB = await import('roslib');

    // Verify the module loaded (CJS->ESM interop worked)
    expect(ROSLIB).toBeDefined();

    // Verify Ros constructor exists (the primary API we use)
    // roslib may export as default or named depending on bundler resolution
    const Ros = ROSLIB.default?.Ros ?? (ROSLIB as Record<string, unknown>).Ros;
    expect(Ros).toBeDefined();
    expect(typeof Ros).toBe('function');
  });

  it('exposes Topic constructor for pub/sub', async () => {
    const ROSLIB = await import('roslib');
    const Topic = ROSLIB.default?.Topic ?? (ROSLIB as Record<string, unknown>).Topic;
    expect(Topic).toBeDefined();
    expect(typeof Topic).toBe('function');
  });
});
```

Note: Uses dynamic `import()` to closely mirror how the app will load roslib at runtime. Tests both `Ros` (connection) and `Topic` (pub/sub) constructors -- the two APIs the transport layer (Phase 5) depends on.

```typescript
import { test, expect } from '@playwright/test';

test.describe('Visual Gate — Phase 01: Scaffolding', () => {
  test('app renders with content (not blank screen)', async ({ page }) => {
    await page.goto('/');

    // Root element exists and has children
    const root = page.locator('#root');
    await expect(root).toBeVisible();

    const appRoot = page.locator('#app-root');
    await expect(appRoot).toBeVisible();

    // App has meaningful content (not blank)
    const heading = page.locator('h1');
    await expect(heading).toHaveText('Robot Telemetry Dashboard');
  });

  test('no console errors on load', async ({ page }) => {
    const consoleErrors: string[] = [];
    page.on('console', (msg) => {
      if (msg.type() === 'error') {
        consoleErrors.push(msg.text());
      }
    });

    await page.goto('/');
    await page.waitForLoadState('networkidle');

    expect(consoleErrors).toEqual([]);
  });

  test('recharts renders SVG chart', async ({ page }) => {
    await page.goto('/');

    // Recharts renders as SVG -- verify the chart panel contains an SVG element
    const chartPanel = page.locator('[data-testid="recharts-panel"]');
    await expect(chartPanel).toBeVisible();

    const svg = chartPanel.locator('svg.recharts-surface');
    await expect(svg).toBeVisible();
  });

  test('react-grid-layout renders grid', async ({ page }) => {
    await page.goto('/');

    // react-grid-layout renders with class "react-grid-layout"
    const grid = page.locator('.react-grid-layout');
    await expect(grid).toBeVisible();

    // Grid has child items
    const items = page.locator('.react-grid-item');
    await expect(items).toHaveCount(2);
  });

  test('baseline screenshot', async ({ page }) => {
    await page.goto('/');
    await page.waitForLoadState('networkidle');

    // Capture baseline screenshot for visual regression
    await expect(page).toHaveScreenshot('01-scaffolding-baseline.png', {
      fullPage: true,
    });
  });
});
```

This satisfies D-14: app renders, root has children, no console errors, `toHaveScreenshot()` baseline. Also verifies Recharts and react-grid-layout render in a real browser (spirit of roadmap success criteria #5) without dedicated smoke test files (per D-11, D-13).
